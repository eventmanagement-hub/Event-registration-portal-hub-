<body>
    <!-- Introduction Page -->
    <div id="introPage">
        <h1>Welcome to the Event Management System</h1>
        <p>Manage and organize events seamlessly with our system.</p>
        <button onclick="goToRegister()">Click Here to Register</button>
    </div>

    <!-- Registration Page -->
    <div id="registerPage" style="display: none;">
        <h1>Event Management System</h1>
        <h2>Register for an Event</h2>
        <form id="registrationForm">
            <label for="fullName">Full Name:</label>
            <input type="text" id="fullName" placeholder="Enter your name" required><br>

            <label for="email">Email Address:</label>
            <input type="email" id="email" placeholder="Enter your email" required><br>

            <label for="eventName">Event Name:</label>
            <input type="text" id="eventName" placeholder="Enter event name" required><br>

            <button type="button" onclick="register()">Register</button>
        </form>
        <button onclick="goToIntro()">Back to Menu</button>

        <h3>All Registrations:</h3>
        <ul id="registrationList"></ul>

        <h2>Feedback Section</h2>
        <form id="feedbackForm">
            <label for="feedback">Your Feedback:</label><br>
            <textarea id="feedback" rows="5" placeholder="Write your feedback"></textarea><br>
            <button type="button" onclick="submitFeedback()">Submit Feedback</button>
        </form>
        <h3>Feedbacks:</h3>
        <ul id="feedbackList"></ul>
    </div>

    <script>
        // Function to navigate to the registration page
        function goToRegister() {
            document.getElementById("introPage").style.display = "none";
            document.getElementById("registerPage").style.display = "block";
        }

        // Function to navigate back to the introduction page
        function goToIntro() {
            document.getElementById("introPage").style.display = "block";
            document.getElementById("registerPage").style.display = "none";
        }

        // Function to handle registration
        function register() {
            const fullName = document.getElementById("fullName").value;
            const email = document.getElementById("email").value;
            const eventName = document.getElementById("eventName").value;

            if (fullName && email && eventName) {
                const registrationList = document.getElementById("registrationList");
                const listItem = document.createElement("li");
                listItem.textContent = `Name: ${fullName}, Email: ${email}, Event: ${eventName}`;
                registrationList.appendChild(listItem);

                alert("Registration Successful!");

                // Clear input fields
                document.getElementById("fullName").value = "";
                document.getElementById("email").value = "";
                document.getElementById("eventName").value = "";
            } else {
                alert("Please fill in all fields.");
            }
        }

        // Function to handle feedback submission
        function submitFeedback() {
            const feedback = document.getElementById("feedback").value;

            if (feedback) {
                const feedbackList = document.getElementById("feedbackList");
                const listItem = document.createElement("li");
                listItem.textContent = feedback;
                feedbackList.appendChild(listItem);

                alert("Feedback Submitted!");

                // Clear feedback field
                document.getElementById("feedback").value = "";
            } else {
                alert("Please write your feedback.");
            }
        }
    </script>
